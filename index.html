function getBestXI() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // ══════════════════════════════════════════ // 1. Đọc DATA_LOG — tính các chỉ số nâng cao theo từng cầu thủ // B=Ngày, D=Tỉ số, E=Tên, J=GF, K=GP(pen thắng), L=AS // O=AG(bàn thua), F=Trạng thái // ══════════════════════════════════════════ const logSheet = ss.getSheetByName("DATA_LOG"); const logData = logSheet.getDataRange().getValues(); // Map ngày → { tySo, tongBanThang, tongBanThua } của trận đó // Đồng thời map tên → danh sách ngày có tham gia const matchByDate = {}; // dateKey → { tyso, ourGoals, theirGoals, players: Set } const playerDates = {}; // ten → [dateKey, ...] for (let i = 1; i < logData.length; i++) { const r = logData[i]; const ngay = r[1]; // Cột B const tyso = String(r[3] || '').trim(); // Cột D: "4 - 1" const ten = String(r[4] || '').trim(); // Cột E const status = String(r[5] || '').trim(); // Cột F if (!ngay || !ten) continue; const dateKey = Utilities.formatDate(new Date(ngay), "Asia/Ho_Chi_Minh", "yyyy-MM-dd"); // Build match info if (!matchByDate[dateKey]) { const parts = tyso.split('-').map(s => parseInt(s.trim()) || 0); matchByDate[dateKey] = { tyso: tyso, ourGoals: parts[0] || 0, theirGoals: parts[1] || 0, players: new Set() }; } if (status === "Tham gia" || status === "Trễ") { matchByDate[dateKey].players.add(ten); if (!playerDates[ten]) playerDates[ten] = []; if (!playerDates[ten].includes(dateKey)) playerDates[ten].push(dateKey); } } // Tính chỉ số nâng cao cho từng cầu thủ const advancedStats = {}; // ten → stats // Đọc lại DATA_LOG để tính GF, AS, AG theo từng trận của cầu thủ // Map: dateKey → { playerGF, playerAS, playerAG } const playerMatchStats = {}; // ten → { dateKey → { gf, gp, ast, ag } } for (let i = 1; i < logData.length; i++) { const r = logData[i]; const ngay = r[1]; const ten = String(r[4] || '').trim(); if (!ngay || !ten) continue; const dateKey = Utilities.formatDate(new Date(ngay), "Asia/Ho_Chi_Minh", "yyyy-MM-dd"); if (!playerMatchStats[ten]) playerMatchStats[ten] = {}; playerMatchStats[ten][dateKey] = { gf: Number(r[9]) || 0, // Cột J gp: Number(r[10]) || 0, // Cột K ast: Number(r[11]) || 0, // Cột L ag: Number(r[14]) || 0 // Cột O }; } // Với mỗi cầu thủ đã tham gia, tính: // - tỉ lệ thắng: số trận thắng (ourGoals > theirGoals) / tổng trận tham gia // - đóng góp bàn thắng: (gf+ast) / tổng ourGoals của các trận đó // - giữ sạch lưới: số trận theirGoals == 0 / tổng trận tham gia // - số trận vắng + số ngày kể từ trận cuối const allDates = Object.keys(matchByDate).sort(); // tất cả ngày thi đấu const today = new Date(); Object.keys(playerDates).forEach(ten => { const dates = playerDates[ten].sort(); let wins = 0, draws = 0, cleanSheets = 0; let totalOurGoals = 0, playerContrib = 0; dates.forEach(dk => { const m = matchByDate[dk]; const ps = (playerMatchStats[ten] || {})[dk] || {}; if (m.ourGoals > m.theirGoals) wins++; else if (m.ourGoals === m.theirGoals) draws++; if (m.theirGoals === 0) cleanSheets++; totalOurGoals += m.ourGoals; playerContrib += (ps.gf || 0) + (ps.ast || 0); }); const totalPlayed = dates.length; const winRate = totalPlayed > 0 ? Math.round((wins / totalPlayed) * 100) : 0; const cleanRate = totalPlayed > 0 ? Math.round((cleanSheets / totalPlayed) * 100) : 0; const contribRate = totalOurGoals > 0 ? Math.round((playerContrib / totalOurGoals) * 100) : 0; // Số trận vắng: số trận trong allDates sau ngày tham gia cuối cùng const lastDate = dates[dates.length - 1]; const missedMatches = allDates.filter(d => d > lastDate); const lastDateObj = new Date(lastDate); const diffDays = Math.floor((today - lastDateObj) / (1000 * 60 * 60 * 24)); advancedStats[ten] = { winRate: winRate, draws: draws, losses: totalPlayed - wins - draws, cleanRate: cleanRate, contribRate: contribRate, missedCount: missedMatches.length, missedDays: diffDays, lastMatch: lastDate }; }); // ══════════════════════════════════════════ // 2. Đọc CAU_THU — stats + bday + avatar // A=ID, B=Họ tên, C=Số áo, D=Vị trí, // E=Trạng thái, F=BDay, G=Benefit, H=Công nợ, // I=Trận, J=MVP, K=TR, L=BM, M=GF, N=GP, // O=AS, P=YC, Q=RC, R=AG, S=AP, T=OG, // U=TR:Trễ, V=Avt // ══════════════════════════════════════════ const cauThuSheet = ss.getSheetByName("CAU_THU"); const cauThuData = cauThuSheet.getDataRange().getValues(); const playerMap = {}; for (let i = 1; i < cauThuData.length; i++) { const r = cauThuData[i]; const ten = String(r[1] || '').trim(); if (!ten) continue; // BDay: chỉ lấy ngày/tháng let bday = ''; if (r[5] instanceof Date) { bday = Utilities.formatDate(r[5], "Asia/Ho_Chi_Minh", "dd/MM"); } playerMap[ten] = { soAo: r[2] || '', bday: bday, tran: Number(r[8]) || 0, mvp: Number(r[9]) || 0, tr: Number(r[10]) || 0, bm: Number(r[11]) || 0, gf: Number(r[12]) || 0, gp: Number(r[13]) || 0, ast: Number(r[14]) || 0, yc: Number(r[15]) || 0, rc: Number(r[16]) || 0, ag: Number(r[17]) || 0, ap: Number(r[18]) || 0, og: Number(r[19]) || 0, treTran: Number(r[20]) || 0, avatar: String(r[21] || '').trim() }; } // ══════════════════════════════════════════ // 3. Đọc POINT // ══════════════════════════════════════════ const pointSheet = ss.getSheetByName("POINT"); const rows = pointSheet.getDataRange().getValues(); const COL_STT = 0; const COL_TEN = 1; const COL_VITRI = 2; const COL_CHUYEN_CAN = 3; const COL_CHUYEN_MON = 4; const COL_HOAT_DONG = 6; const COL_DIEM = 7; // ══════════════════════════════════════════ // 4. Formation // ══════════════════════════════════════════ const STARTING = { "Thủ môn": 1, "Hậu vệ cánh": 2, "Trung vệ": 2, "Tiền vệ TT": 3, "Tiền vệ cánh": 2, "Tiền đạo": 1 }; const BENCH_COUNT = { "Thủ môn": 1, "Hậu vệ cánh": 2, "Trung vệ": 1, "Tiền vệ TT": 2, "Tiền vệ cánh": 2, "Tiền đạo": 2 }; // ══════════════════════════════════════════ // 5. Build player objects // ══════════════════════════════════════════ const byPos = {}; for (let i = 1; i < rows.length; i++) { const r = rows[i]; if (!r[COL_TEN]) continue; const ten = String(r[COL_TEN]).trim(); const pos = String(r[COL_VITRI]).trim(); const extra = playerMap[ten] || {}; const adv = advancedStats[ten] || {}; if (!byPos[pos]) byPos[pos] = []; byPos[pos].push({ stt: String(r[COL_STT]).trim(), ten: ten, viTri: pos, chuyenCan: Number(r[COL_CHUYEN_CAN]) || 0, chuyenMon: Number(r[COL_CHUYEN_MON]) || 0, hoatDong: Number(r[COL_HOAT_DONG]) || 0, diem: Number(r[COL_DIEM]) || 0, // CAU_THU stats soAo: extra.soAo || '', bday: extra.bday || '', avatar: extra.avatar || '', tran: extra.tran || 0, mvp: extra.mvp || 0, bm: extra.bm || 0, gf: extra.gf || 0, gp: extra.gp || 0, ast: extra.ast || 0, yc: extra.yc || 0, rc: extra.rc || 0, ag: extra.ag || 0, ap: extra.ap || 0, og: extra.og || 0, treTran: extra.treTran || 0, // Advanced stats từ DATA_LOG winRate: adv.winRate ?? null, cleanRate: adv.cleanRate ?? null, contribRate: adv.contribRate ?? null, missedCount: adv.missedCount ?? 0, missedDays: adv.missedDays ?? 0 }); } // ══════════════════════════════════════════ // 6. Sắp xếp & tách chính thức / dự bị // ══════════════════════════════════════════ const squad = {}; const bench = {}; Object.keys(STARTING).forEach(pos => { const sorted = (byPos[pos] || []).sort((a, b) => b.diem - a.diem); const s = STARTING[pos]; const b = BENCH_COUNT[pos] || 0; squad[pos] = sorted.slice(0, s); bench[pos] = sorted.slice(s, s + b); }); return { updatedAt: Utilities.formatDate(new Date(), "Asia/Ho_Chi_Minh", "dd/MM/yyyy HH:mm"), formation: "4-5-1", squad: squad, bench: bench }; }
